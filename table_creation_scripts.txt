-- Create Database
CREATE DATABASE budget-buddy;

-- Create USERS table
CREATE TABLE USERS (
    USERID INT PRIMARY KEY,
    USERNAME VARCHAR(50) NOT NULL,
    PASSCODE VARCHAR(50) NOT NULL,
    EMAIL VARCHAR(75) NOT NULL
);

-- Create EXPENSE table
CREATE TABLE EXPENSE (
    EXPENSEID INT PRIMARY KEY,
    EXPENSEBUDGETID INT,
    EXPENSECATEGORYID INT,
    EXPENSENAME VARCHAR(50) NOT NULL,
    EXPENSEAMOUNT DECIMAL(10, 2) NOT NULL,
    EXPENSEDATE DATE NOT NULL,
    EXPENSEDESCRIPTION VARCHAR(50),
    FOREIGN KEY (EXPENSEBUDGETID) REFERENCES BUDGET(BUDGETID),
    FOREIGN KEY (EXPENSECATEGORYID) REFERENCES CATEGORY(CATEGORYID)
);

-- Create BUDGET table
CREATE TABLE BUDGET (
    BUDGETID INT PRIMARY KEY,
    BUDGETNAME VARCHAR(50) NOT NULL,
    STARTDATE DATE NOT NULL,
    ENDDATE DATE,
    TOTALAMOUNT DECIMAL(10, 2),
    BUDGETUSERID INT,
    FOREIGN KEY (BUDGETUSERID) REFERENCES USERS(USERID)
);

-- Create CATEGORY table
CREATE TABLE CATEGORY (
    CATEGORYID INT PRIMARY KEY,
    CATEGORYNAME VARCHAR(50) NOT NULL
);
